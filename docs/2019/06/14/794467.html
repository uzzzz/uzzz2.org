载入数据集

df = pd.DataFrame({
    '学生姓名':['小红','小花','小明','小军','小雷'],
    '性别':['女','女','男','男','男'],
    '语文成绩':[58,78,96,64,85],
    '数学成绩':[90,69,45,35,57],
    '英语成绩':[68,80,46,66,96]
})
