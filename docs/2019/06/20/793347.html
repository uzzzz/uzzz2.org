#encoding:utf-8
"""
例3：小精灵：你好，欢迎古灵阁，请问您需要帮助吗？需要or不需要？
你：需要
小精灵：请问你需要什么帮助呢？
1：存款
2：货币兑换
3：咨询
你：2
小精灵：金加隆和人名币的兑换率为1：51.3，即1金加隆=51.3人名币
小精灵：好的，我知道了，你需要兑换金加隆
你：输入数字N
小精灵：你需要给我N*51.3人名币
注：1.如果你不需要帮助，小精灵会礼貌的说“再见”
    2.补充1和3的交互
"""
needORno=input("你好，欢迎古灵阁，请问您需要帮助吗？需要or不需要？")
if needORno=='不需要':
    print("再见")
elif needORno=='需要':
    need=int(input("请问你需要什么帮助呢？1：存款2：货币兑换3：咨询"))
    if int(need)==2:
        print("金加隆和人名币的兑换率为1：51.3，即1金加隆=51.3人名币")
        print("好的，我知道了，你需要兑换金加隆")
        youPut=int(input())
        print("你需要给我%f人民币"%(youPut*51.3))
    elif  int(need)==1:
        print("金加隆和人名币的兑换率为1：51.3，即1金加隆=51.3人名币")
        print("好的，我知道了，你需要存金加隆")
        youGive = int(input())
        print("你需要给我%f人民币" % (youGive * 51.3))
        print("你已经存储了%f金加隆") % youGive
    if int(need)==3:
        print("金加隆和人名币的兑换率为1：51.3，即1金加隆=51.3人名币")
