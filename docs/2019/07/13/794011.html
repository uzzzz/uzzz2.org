ELK-7.2.0集群搭建-elasticsearch

cd /home/kibana-7.2.0-linux-x86_64
vi config/kibana.yml

#端口
server.port: 5601
#服务Ip
server.host: "192.168.25.180"
#服务名
server.name: "kibana"
#ES地址
elasticsearch.hosts: ["http://192.168.25.180:9200"]
su bigdata

bin/kibana

后台启动

nohup bin/kibana &
