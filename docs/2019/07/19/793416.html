安装源

#阿里centos7
wget http://mirrors.aliyun.com/repo/Centos-7.repo
#网易centos7
wget http://mirrors.163.com/.help/CentOS7-Base-163.repo
#清理
yum clean all
#生成缓存
yum makecache
#更新系统
yum -y update

#使用YUM查找软件包
yum search ~
#列出所有可安装的软件包
yum list
#列出所有可更新的软件包
yum list updates
#列出所有已安装的软件包
yum list installed
#列出所有已安装但不在Yum Repository內的软件包
yum list extras
#列出所指定软件包
yum list ～
#使用YUM获取软件包信息
yum info ～
#列出所有软件包的信息
yum info
#列出所有可更新的软件包信息
yum info updates
#列出所有已安裝的软件包信息
yum info installed
#列出所有已安裝但不在Yum Repository內的软件包信息
yum info extras
#列出软件包提供哪些文件
yump rovides ~
