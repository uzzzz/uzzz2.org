生成MD5：

import hashlib
import json

def get_md5(data):
    m5 = hashlib.md5()
    m5.update(json.dumps(data).encode())
    return m5.hexdigest()

print(get_md5("hello")) # 5deaee1c1332199e5b5bc7c5e4f7f0c2

生成SHA1：

import hashlib
import json

def get_sha1(data) :
    s1 = hashlib.sha1()
    s1.update(json.dumps(data).encode())
    return  s1.hexdigest()

print(get_sha1("hello")) # a1f2fbfe2c4ad81749cd0380b735295d06f9d0c4
