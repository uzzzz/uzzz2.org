2，我们来举一个简单例子来说明一下：
代码实现如下：simplesvm.py

from sklearn import svm
x = [[2, 0], [1, 1], [2, 3]]
y = [0, 0, 1]
#调用svm的分类器函数
clf = svm.SVC(kernel = 'linear')
#建立模型
clf.fit(x, y)
print(clf)
# get support vectors
print( clf.support_vectors_)
# get indices of support vectors
print( clf.support_)
# get number of support vectors for each class
print (clf.n_support_)
